股票数据分析定向爬虫

注：此代码仅供学习和参考

需求：

1：通过一个网站获取A股股票代码
2：通过A股股票代码在另一个网站获取股票的部分具体信息

明确目标：

1：搜索股票代码一览表，在查询多个网站后，发现从连板网获取A股股票代码更为简单，可以查询按地区分的股票代码：
https://www.banban.cn/gupiao/ + 地址
2：查询股票信息，共有两个选择，东方财富网或者雪球网，此处使用了雪球网。
通过查看网页源代码，决定在雪球网，获取具体信息为当前价格，和涨跌情况
url分为两类，SZ或SH+股票代码
https://xueqiu.com/S/SZ
https://xueqiu.com/S/SH

技术路线：

1.requests+BeautifulSoup+re
2.Scrapy框架
此处暂时采取第一种技术路线

    任意点开一个股票信息，查看源代码（不是检查代码），发现其较为复杂，且不便分析，

    故我选择直接用程序爬取后规范化输出的数据，并结合网页源代码，作为分析，

    html = get_html(inf_url, head='')
soup = BeautifulSoup(html,'html.parser')
print(soup.prettify())#规范化输出，方便分析网页

    程序结构：

    获取网页源代码 get_html(url,head)
    获得股票代码 Stock_Code(html,Stock_List)
    通过股票代码，跳转个股信息，爬取当前价格和涨跌 Stock_inf(Stock_List,Stock_List_Inf,place)
    主函数：选择爬取范围

    完整代码（附注释）

    import requests
from bs4 import BeautifulSoup
import bs4
import re
import traceback #输出报错信息
import time

def get_html(url,head):

    html = requests.get(url,headers=head) #注意爬取head
    html.raise_for_status()  # 网页访问状态错误，则产生异常
    html.encoding = html.apparent_encoding  #此处可以人工优化
    return html.text

#获得股票代码
#以字典格式，返回股票代码列表
def Stock_Code(html,Stock_List):
    soup = BeautifulSoup(html,'html.parser')
    a = soup.find_all('a')#所有'a'标签
    # print(a)
    for i in a:
        try:
            # print(i)
            href = i.string#获取其非属性字符串
            # print(href)
            s = re.findall(r'.+?\(\d{6}\)',href) #直接获取类似于外高桥(600648)这种格式的字符串
            #返回的是列表！！！
            if len(s)!=0:
                # print(s)
                Stock_List[s[0].split('(')[0]] = s[0].split('(')[1].replace(')','')
                # print(s[0].split('(')[0],":",s[0].split('(')[1].replace(')',''))
        except:
                continue

#子函数里一次性查询所有股票
#获得股票当前价格和成交量
#返回具体信息
def Stock_inf(Stock_List,Stock_List_Inf,place):
    lst = len(Stock_List) #总共爬取多少支股票
    print("总共爬取{}股票共{}支".format(place,lst))
    count = 0
    t = time.localtime()
    t = time.strftime("%Y_%m_%d_%H_%M", t)
    fpath = t+'股票数据.txt'
    with open(fpath, 'a', encoding='utf-8') as f:
        f.write('{}股票数据\n'.format(place))
        f.write('股票名称\t股票代码\t当前价格\t\t涨跌情况\n')
    #遍历股票代码
    for stock,code in Stock_List.items():
        try:
            inf_url = "https://xueqiu.com/S/SZ" + code
            html = get_html(inf_url,head = '')
        except:
            inf_url = "https://xueqiu.com/S/SH" + code
            html = get_html(inf_url, head='')
        soup = BeautifulSoup(html,'html.parser')
        # print(soup.prettify())#规范化输出，方便分析网页
        # print("********************************************")
        # 拥有class=stock-current的div标签在其子孙节点里
        for s in soup.body.descendants:
            # 注意find方法的使用，由于class是关键字，无法直接使用，故使用class_
            #s.find(class_="stock-current")
            value = s.find(attrs={'class': "stock-current"})
            up_down = s.find(attrs={'class': "stock-change"})
            if up_down != None:
                up_down = str(up_down.string)
            if  value != None:
                value = str(value.strong.string)#获取了当天价格
            if (up_down!=None) and (value!=None):
                Stock_List_Inf[stock] = [code,value,up_down]
                st = stock+'\t'+code+'\t'+value+'\t'+up_down+'\n' #方便写入文件
                with open(fpath, 'a',encoding='utf-8') as f:
                    f.write(st)
                count = count+1
                bar = '['+'*'*int(count*100/lst)+']'#多行进度条暂时不会,似乎有点bug，有时会跳
                print('\r当前进度{:.2f}%\t正在爬取：{}\t{}'.format(count*100/lst,stock,bar),end='')#\r 表示将光标的位置回退到本行的开头位置
                break
    return Stock_List_Inf

def main():
    place = {"合肥":"anhui/hefei.html",'上海':"shanghai/"}
    # place = {"合肥": "anhui/hefei.html"} #可以选择自己加上
    # places = place["合肥"]
    for pl in list(place.keys()):
        Stock_List = {}  # 存放股票代码的字典
        Stock_List_Inf = {} #最终存放股票信息的地方
        print('正在爬取{}的股票'.format(pl))
        Code_Url = "https://www.banban.cn/gupiao/"+ place[pl]#此处仅以上海为例
        html = get_html(Code_Url,'')#此网站可以不用加header
        Stock_Code(html,Stock_List)#获取股票代码和名称
        Stock_inf(Stock_List,Stock_List_Inf,pl)#获得股票数据具体信息
        print('\n{}股票爬取完成'.format(pl))
    print('股票爬取完成')


if __name__ == '__main__':
    main()
#traceback.print_exc() #打印异常信息
